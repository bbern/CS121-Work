#CountOdds Function

countOdds <- function(x) {

  • k<-0
  • for(n in x) {
  • if (n%%2==1)k<-k+1
  • }
  • k
  • }

#CountEvens Function countEvens <-function(x) {

  • y <-0
  • for (n in x) {
  • if(n%%2==0)y <-y+1
  • }
  • y
  • }
